ค่า null,undefined และ NaN

null คือ ตัวแปรที่ไม่มีค่าใด ๆ เลย ไม่เท่ากับ 0 และไม่เท่ากับตริงว่าง ไม่สามารถนำไปคำนวณใดๆ ได้ แต่หากนำไปเปรียบเทียบด้วยเงื่อนไขจะมีค่าเท่ากับค่า false

let a = null ;
if (!a) {
   alert (" a is null ");
} else {
   alert (" a is not null ") ;
}

undefined คือ ตัวแปรที่ประกาศเอาไว้แต่ไม่ได้กำหนดค่าใด ๆ ให้กับมัน ยกตัวอย่าง เช่น

let a ;
alert(a) ;

NaN (Not a Number) คือ การนำตัวแปรที่ไม่ใช่ตัวเลขไปคำนวณทางคณิตศาสตร์

let a = 10 ;
let b = "x" ;
alert(10-b) ;
